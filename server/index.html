Endpoints

Get users

type:

GET

url:

http://localhost:8000/users?search=${search}&sort=${sort}&page=${page}&quantity=${quantity}

response:

{
  users: { id: number, name: string, phoneNumber: string [] } []
  length: number
}

params:

search - string
sort - null | 'desc' | 'asc'
page - number
quantity - number

Get user by id

type:

GET

url:

http://localhost:8000/getuserbyid/${id}

response:

{
  id: number,
  name: string,
  phoneNumber: string []
}

Add user

type:

POST

url:

http://localhost:8000/adduser

body:

{
  name: string,
  phoneNumber: string []
}

response:

string

Update user

type:

PUT

url:

http://localhost:8000/updateuser/${id}

body:

{
  name: string,
  phoneNumber: string []
}

response:

string

Delete user

type:

DELETE

url:

http://localhost:8000/deleteuser/${id}

response:

string
